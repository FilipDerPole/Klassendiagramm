
public class Fach {

		name = nam;
		wochensttunden = wochenst;
		raumanforderungen = raumanf;
	}

	public boolean addLehrer(Lehrer neuLehrer){
		try{	
			LehrerListe.add(neuLehrer);
		}				
			
			catch(Exception e){

		}
		return true;
		
	}

		return raumanforderung;

		raumanforderung = raumanf;
